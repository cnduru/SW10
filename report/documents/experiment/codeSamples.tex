            throw new Exception();

    private void processVerifyPIN() throws Exception
    {
        if(callId != 2)
        {
            throw new Exception();
        }

        int pinLength = 4;
        int faultCode = 255;
        int triesRemaining;

        callId = 3;

        short count = setIncomingAndReceive();    // get expected data

        if(callId != 4)
        {
            throw new Exception();
        }

        if(count < pinLength) throw new Exception();

        callId = 4;

        if(isInvalid() != false)
        {
            if(callId != 5)
            {
                throw new Exception();
            }

            callId = 5;
            triesRemaining = getTriesRemaining();
            
            if(callId != 6)
            {
                throw new Exception();
            }

            throw new Exception();
        }

        callId = 2;
    }


    private boolean isInvalid() throws Exception
    {
        if(callId != 4)
        {
            throw new Exception();
        }

        callId = 5;

        return true;
    }

    private short setIncomingAndReceive() throws Exception
    {
        if(callId != 3)
        {
            throw new Exception();
        }

        callId = 4;

    }

    {
        if(callId != 5)
        {
            throw new Exception();
